python3 -m twine upload dist/*0.0.5* --verbose

# use this to install from pypitest
python3 -m pip install --index-url https://test.pypi.org/simple/ --no-deps BART_SURVIVAL==0.0.1.3
pip install  BART_SURVIVAL==0.0.5

# use this to build docs
python3 -m build
